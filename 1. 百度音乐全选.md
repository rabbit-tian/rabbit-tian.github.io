1. 具体的代码实现

   ```javascript
   html结构：

   	<div class="wrap" id="app">
   		<div class="baidu">
   			<ul class="list list-head">
   				<li>
   					<div>
   						<input type="checkbox" v-model="isAllChecked" />全选
   					</div>
   					<span>歌单</span>
   					<span>歌手</span>
   					<span>专辑</span>
   				</li>
   			</ul>
   			<ul class="list list-body">
   				<li :class="{checkedColor:value.checked}" v-for="(value,index) in list">
   					<div><input type="checkbox" v-model="value.checked"></div>
   					<span>{{value.song}}</span>
   					<span>{{value.name}}</span>
   					<span>{{value.album}}</span>
   				</li>
   			</ul>
   			<div class="select">
   				<span class="selectAll">

   						<span>统计：</span>
   				</span>
   				<div class="others">
   					<span><em></em>歌手有：{{totalMan}}位</span>
   					<span><em></em>专辑有{{totalNum}}</span>
   				</div>
   			</div>
   		</div>
   	</div>

   js 代码块

   <script>
   		let data = [{
   				id: Date.now() + Math.random(),
   				name: '邓紫棋',
   				song: '泡沫',
   				checked: false,
   				album: 10
   			},
   			{
   				id: Date.now() + Math.random(),
   				name: '王杰',
   				song: '泡沫',
   				checked: false,
   				album: 10
   			},
   			{
   				id: Date.now() + Math.random(),
   				name: '邓紫棋',
   				song: '泡沫',
   				checked: false,
   				album: 10
   			}
   		]

   		let vm = new Vue({
   			el: '#app',
   			data: {
   				list: data
   			},
   			computed: {
   				isAllChecked: {
   					get() { // 取值的时候触发
   						let newList = this.list.filter((item) => {
   							return item.checked
   						})
   						return newList.length === this.list.length
   					},
   					set(newValue) {//设置值作为参数传过来
   						this.list.forEach((item) => {// 设置值触发
   							item.checked = newValue
   						})
   					}
   				},
   				totalNum() {
   					let total = this.list.reduce((item1, item2) => {
   						return item1 + item2.album
   					}, 0)

   					// console.log(total);

   					return total
   				},
   				totalMan() {

   					return this.list.length
   				}

   			}
   		})
   	</script>
   ```
